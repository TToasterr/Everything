def cmd(message, msg, invitelink, me):
  global doMsgOut, msgOut, consoleOut, channel, msgAmm

  try:
    args = msg[12:]
    args = args.split(", ")
    input = args[0]
    spaceAmm = args[1]
  except:
    doMsgOut = True
    msgOut = ["You didnt use enough arguments!"]
    consoleOut = "%s tried to vaporwave but didn't use enough args." % (message.author)
    channel = [message.channel]
    msgAmm = 1
    return()

  final = []

  for char in input:
    final.append(char)

  spaces = " " * int(spaceAmm)

  final = spaces.join(final)

  doMsgOut = True
  msgOut = [final]
  consoleOut = "%s vaporwaved '%s'." % (message.author, input)
  channel = [message.channel]
  msgAmm = 1
